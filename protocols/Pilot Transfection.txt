<transfection_protocol>
  <title>HEK293 Transfection using Viafect</title>
  <goal>Introduce plasmid DNA (DI-GFP + Cre, CMV-mCherry, or Cre control) into HEK293 cells.</goal>

  <materials>
    <item>Micropipettes (P20, P200) and tips</item>
    <item>3x 0.5mL Microcentrifuge tubes</item>
    <item>Centrifuge</item>
    <item>Vortex</item>
    <item>Viafect Transfection Reagent</item>
    <item>Analytical Balance</item>
    <item>OptiMEM (Reduced Serum Medium)</item>
    <item>DMEM + 10% FCS (without Pen/Strep)</item>
    <item>Plasmids: DI-GFP + Cre, CMV-mCherry, and Cre (control)</item>
    <item>1X PBS (Sterile)</item>
  </materials>

  <reaction_mix_per_well>
    <setup plate_type="24-well">
      <reagent name="OptiMEM">40 uL</reagent>
      <reagent name="Viafect">2.0 uL</reagent>
      <reagent name="pDNA">~1.0 ug</reagent>
    </setup>
    <setup plate_type="12-well">
      <reagent name="OptiMEM">80 uL</reagent>
      <reagent name="Viafect">4.0 uL</reagent>
      <reagent name="pDNA">~1.0 - 2.0 ug</reagent>
    </setup>
  </reaction_mix_per_well>

  <protocol_steps>
    <step id="1" title="Cell Wash">
      Wash cells in each well with sterile 1X PBS. Carefully leave a thin layer of fluid over cells to prevent detachment.
    </step>

    <step id="2" title="Media Replacement">
      Replace PBS with fresh DMEM + 10% FCS (without P/S). Use 0.5 mL for a 24-well plate or 1.0 mL for a 12-well plate.
    </step>

    <step id="3" title="Tube Preparation">
      Acquire 3x 0.5mL centrifuge tubes (one for each plasmid substrate) and record their empty weight on the analytical balance.
    </step>

    <step id="4" title="Complex Formation">
      In the centrifuge tubes, combine OptiMEM, Viafect, and pDNA according to the "reaction_mix_per_well" table. Prepare one tube for each condition: DI-GFP + Cre, CMV-mCherry, and Cre control.
    </step>

    <step id="5" title="Complex QC">
      Weigh the final weight of each centrifuge tube containing the transfection complex and record the value.
    </step>

    <step id="6" title="Incubation">
      Incubate the transfection mixture at Room Temperature (RT) for 15–30 minutes (alternatively, 5 minutes if using a rapid protocol).
    </step>

    <step id="7" title="Transfection">
      Add the transfection mixture to the cells dropwise and slowly to ensure even distribution and minimal cell stress.
    </step>

    <step id="8" title="Final Incubation">
      Incubate the cells for 1–2 days (overnight is usually sufficient). Raise hand to signify completion of the setup.
    </step>
  </protocol_steps>

  <common_errors>
    <error>Accidentally drying out the cell monolayer during the PBS wash step.</error>
    <error>Using Media with Antibiotics (P/S) during transfection, which can significantly decrease cell viability.</error>
  </common_errors>

  <notes>
    <note>The pDNA amounts are listed in micrograms (ug), though the source text used "mg" (standard transfections typically use ug/well).</note>
    <note>Ensure Viafect is flicked or lightly vortexed before use as per manufacturer's suggestion.</note>
  </notes>
</transfection_protocol>
